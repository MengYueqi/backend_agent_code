题目 6：实现快排和堆排

题目描述

给定一个整数切片 `nums`，请分别实现快速排序和堆排序，将其按升序排列。

你需要提供两个函数：一个使用快排思想完成排序，另一个使用堆排思想完成排序。

题目重点不是调用标准库排序，而是手写出这两种经典排序的核心过程，并正确处理常见边界情况。

具体要求

1. 需要实现两个函数：`QuickSort(nums []int)` 和 `HeapSort(nums []int)`。

2. 两个函数都应直接修改输入切片，使其最终按升序排列，不要求返回新切片。

3. 当输入为 `nil`、空切片或仅包含一个元素时，函数应直接返回，不应发生 panic。

4. 输入数据可能包含重复元素、负数、零，以及已经有序或完全逆序的情况。

5. 不允许直接调用 `sort.Ints` 或其它现成排序函数完成核心逻辑。

6. 排序结果只要求数值升序，不要求稳定性。

输入输出说明

  • 输入：`nums []int`，待排序的整数切片。
  • 输出：无返回值，函数执行结束后，`nums` 应被原地修改为升序结果。

建议函数签名：


func QuickSort(nums []int)
func HeapSort(nums []int)

示例

示例 1（正常场景）

  • 输入：`nums = [5, 1, 4, 2, 8]`
  • 输出：`[1, 2, 4, 5, 8]`
  • 说明：两个函数都应把切片原地排成升序。

示例 2（包含重复元素）

  • 输入：`nums = [3, 1, 2, 3, 2]`
  • 输出：`[1, 2, 2, 3, 3]`
  • 说明：重复元素需要被正确保留并排序。

示例 3（边界：空切片）

  • 输入：`nums = []`
  • 输出：`[]`
  • 说明：空切片直接返回即可。

示例 4（边界：单元素）

  • 输入：`nums = [7]`
  • 输出：`[7]`
  • 说明：单元素切片天然有序。

追问（可选）

  • 如果要降低快排在近乎有序数据上的最坏情况风险，你会如何选择 pivot？
  • 如果要把堆排改成降序，最小改动应放在哪一层逻辑？

package src

// 题目：实现快排和堆排

func partition(nums []int, left, right int) {
	if left >= right {
		return
	}
	pivot := nums[left]
	first, last := left+1, right
	for {
		for first <= last && nums[first] <= pivot {
			first++
		}
		for first <= last && nums[last] >= pivot {
			last--
		}
		if first >= last {
			break
		}
		nums[first], nums[last] = nums[last], nums[first]
	}
	nums[left], nums[last] = nums[last], nums[left]
	partition(nums, left, last-1)
	partition(nums, last+1, right)
}

func siftDown(nums []int, node, last int) {
	for {
		child := node*2 + 1
		if child > last {
			break
		}
		next := node
		if nums[node] < nums[child] {
			next = child
		}
		if child+1 <= last && nums[next] < nums[child+1] {
			next = child + 1
		}
		if next == node {
			return
		} else {
			nums[node], nums[next] = nums[next], nums[node]
			node = next
		}
	}
}

func HeapSort(nums []int) {
	// TODO: implement heap sort
	for i := len(nums) / 2; i >= 0; i-- {
		siftDown(nums, i, len(nums)-1)
	}

	for i := len(nums) - 1; i >= 0; i-- {
		nums[0], nums[i] = nums[i], nums[0]
		siftDown(nums, 0, i-1)
	}

}

package test

import (
	"reflect"
	"testing"

	"go_coding_noalgorithm/src"
)

func cloneInts(nums []int) []int {
	if nums == nil {
		return nil
	}

	got := make([]int, len(nums))
	copy(got, nums)
	return got
}

func TestQuickSort(t *testing.T) {
	tests := []struct {
		name string
		nums []int
		want []int
	}{
		{
			name: "basic case",
			nums: []int{5, 1, 4, 2, 8},
			want: []int{1, 2, 4, 5, 8},
		},
		{
			name: "contains duplicates",
			nums: []int{3, 1, 2, 3, 2},
			want: []int{1, 2, 2, 3, 3},
		},
		{
			name: "contains negative numbers",
			nums: []int{-3, 0, 2, -1, 5},
			want: []int{-3, -1, 0, 2, 5},
		},
		{
			name: "already sorted",
			nums: []int{1, 2, 3, 4, 5},
			want: []int{1, 2, 3, 4, 5},
		},
		{
			name: "reverse sorted",
			nums: []int{9, 7, 5, 3, 1},
			want: []int{1, 3, 5, 7, 9},
		},
		{
			name: "single element",
			nums: []int{7},
			want: []int{7},
		},
		{
			name: "empty slice",
			nums: []int{},
			want: []int{},
		},
		{
			name: "nil slice",
			nums: nil,
			want: nil,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			got := cloneInts(tt.nums)

			src.QuickSort(got)

			if !reflect.DeepEqual(got, tt.want) {
				t.Fatalf("QuickSort(%v) = %v, want %v", tt.nums, got, tt.want)
			}
		})
	}
}
