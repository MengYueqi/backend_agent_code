题目 5：查询是否存在只修改一个字母即可匹配的单词

题目描述

给定一个字符串数组 `words`，数组中每个元素都是一个仅包含小写英文字母的单词。

再给定一个查询单词 `x`，请判断 `words` 中是否存在某个单词，可以通过把 `x` 中的**恰好一个字符替换为另一个字符**得到。

如果存在这样的单词，返回 `true`；否则返回 `false`。

具体要求

1. 只允许执行一次“替换一个字符”的操作，不能插入字符，也不能删除字符。

2. 只有长度与 `x` 相同的单词才可能匹配。

3. 如果 `words` 中存在任意一个满足条件的单词，返回 `true`。

4. 如果没有任何单词满足条件，返回 `false`。

5. 当 `words` 为空时，直接返回 `false`。

6. 如果 `x` 与某个单词完全相同，不算匹配，因为题目要求必须恰好修改一个字符。

输入输出说明

  • 输入：

- `words []string`：候选单词列表，每个单词只包含小写英文字母。

- `x string`：待查询单词，只包含小写英文字母。

  • 输出：`bool`，表示是否存在一个单词能由 `x` 修改恰好一个字符得到。

建议函数签名：


func HasOneLetterDifference(words []string, x string) bool

示例

示例 1（正常场景）

  • 输入：`words = ["hello", "world"], x = "hhllo"`
  • 输出：`true`
  • 说明：`"hhllo"` 只需要把第 2 个字符改为 `'e'`，即可得到 `"hello"`。

示例 2（正常场景）

  • 输入：`words = ["hello", "world"], x = "hello"`
  • 输出：`false`
  • 说明：虽然 `"hello"` 在 `words` 中存在，但没有发生字符修改，不满足“恰好修改一个字符”。

示例 3（正常场景）

  • 输入：`words = ["judge", "badge", "ledge"], x = "budge"`
  • 输出：`true`
  • 说明：`"budge"` 把第 1 个字符从 `'b'` 改为 `'j'` 后，可以匹配 `"judge"`。

示例 4（边界场景）

  • 输入：`words = [], x = "abc"`
  • 输出：`false`
  • 说明：候选列表为空时，不存在任何可匹配项。

追问（可选）

  • 如果 `words` 很大，并且会有很多次查询，如何设计预处理结构来降低单次查询时间？
  • 如果字符集不再局限于小写英文字母，而是任意 Unicode 字符，比较逻辑需要注意什么？

package src

// 题目：查询是否存在只修改一个字母即可匹配的单词
type Trie struct {
	next   []*Trie
	isLast bool
}

func initTrie() Trie {
	return Trie{
		isLast: false,
		next:   make([]*Trie, 26),
	}
}

func (this *Trie) addWord(word string) {
	curNode := this
	for i := 0; i < len(word); i++ {
		if curNode.next[word[i]-'a'] == nil {
			curNode.next[word[i]-'a'] = &Trie{
				isLast: false,
				next:   make([]*Trie, 26),
			}
			curNode = curNode.next[word[i]-'a']
		} else {
			curNode = curNode.next[word[i]-'a']
		}
	}
	curNode.isLast = true
}

func (this *Trie) findWord(word string) bool {
	curNode := this
	for i := 0; i < len(word); i++ {
		if curNode.next[word[i]-'a'] == nil {
			return false
		}
		curNode = curNode.next[word[i]-'a']
	}
	return curNode.isLast
}

func HasOneLetterDifference(words []string, x string) bool {
	// TODO: implement
	trie := initTrie()
	for i := 0; i < len(words); i++ {
		for j := 0; j < len(words[i]); j++ {
			prevWord := []byte(words[i])
			for k := 0; k < 26; k++ {
				prevWord[j] = byte('a' + k)
				if string(prevWord) != words[i] {
					trie.addWord(string(prevWord))
				}
			}
		}
	}
	return trie.findWord(x)
}

test/5_one_letter_match_test.go 
package test

import (
	"testing"

	"go_coding_noalgorithm/src"
)

func TestHasOneLetterDifference(t *testing.T) {
	tests := []struct {
		name  string
		words []string
		x     string
		want  bool
	}{
		{
			name:  "match by replacing one character",
			words: []string{"hello", "world"},
			x:     "hhllo",
			want:  true,
		},
		{
			name:  "exact same word does not count",
			words: []string{"hello", "world"},
			x:     "hello",
			want:  false,
		},
		{
			name:  "different length cannot match",
			words: []string{"abcd", "xy"},
			x:     "abc",
			want:  false,
		},
		{
			name:  "multiple candidates any one match is enough",
			words: []string{"judge", "badge", "ledge"},
			x:     "budge",
			want:  true,
		},
		{
			name:  "needs more than one replacement",
			words: []string{"abc", "def"},
			x:     "axf",
			want:  false,
		},
		{
			name:  "empty dictionary",
			words: []string{},
			x:     "abc",
			want:  false,
		},
		{
			name:  "single character word can match",
			words: []string{"a", "b"},
			x:     "c",
			want:  true,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			got := src.HasOneLetterDifference(tt.words, tt.x)
			if got != tt.want {
				t.Fatalf("HasOneLetterDifference(%v, %q) = %v, want %v", tt.words, tt.x, got, tt.want)
			}
		})
	}
}
