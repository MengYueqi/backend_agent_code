使用给定数字拼出小于n的最大数

题目 4：使用给定数字拼出小于 n 的最大数

题目描述

给定一个非负整数 `n` 和一个数字数组 `digits`（元素范围在 `0~9`），你可以从 `digits` 中反复取数字（每个数字可重复使用任意次）来拼接十进制数。

请返回**严格小于 `n`** 的最大可拼接结果。

具体要求

1. 你只能使用 `digits` 中出现过的数字进行拼接。

2. 每个数字可以被重复使用任意次。

3. 拼接出的结果必须满足 `x < n`。

4. 结果不能有前导零；唯一例外是数字 `0` 本身（即 `x = 0` 合法）。

5. 如果无法拼出任何满足条件的数字，返回 `-1`。

6. `digits` 可能有重复项，重复项不应影响结果（按集合语义处理即可）。

输入输出说明

  • 输入：

- `n int`：目标上界（非负整数）。

- `digits []int`：可用数字集合，元素范围 `0~9`。

  • 输出：`int`，表示可拼出的最大且小于 `n` 的数字；无解返回 `-1`。

建议函数签名：


func BuildMaxLessThanN(n int, digits []int) int

示例

示例 1（正常场景）

  • 输入：`n = 345, digits = [1,3,5]`
  • 输出：`335`
  • 说明：`345` 不可用（含数字 `4`），在可拼数字中 `335` 是小于 `345` 的最大值。

示例 2（正常场景）

  • 输入：`n = 572, digits = [2,5,7]`
  • 输出：`557`
  • 说明：`572` 本身不可取（必须严格小于），最大可行值为 `557`。

示例 3（边界：仅 0 可用）

  • 输入：`n = 10, digits = [0]`
  • 输出：`0`
  • 说明：`0` 合法且小于 `10`，不能构造其它合法数字。

示例 4（边界：无解）

  • 输入：`n = 1, digits = [9]`
  • 输出：`-1`
  • 说明：无法拼出小于 `1` 的合法数字。

追问（可选）

  • 如果 `n` 非常大（超出 `int`），你会如何改造输入类型与比较逻辑？
  • 如果改成“每个数字最多使用 `k` 次”，算法应如何调整？

package src

import "sort"

func findSame(n int, nums []int) bool {
	for i := 0; i < len(nums); i++ {
		if n == nums[i] {
			return true
		}
	}
	return false
}

func findSmall(n int, nums []int) bool {
	for i := 0; i < len(nums); i++ {
		if n > nums[i] {
			return true
		}
	}
	return false
}

func BuildMaxLessThanN(n int, digits []int) int {
	// TODO: implement
	if n < 10 && !findSmall(n, digits) || len(digits) == 0 {
		return -1
	}
	n -= 1
	nums := make([]int, 0)
	for n != 0 {
		nums = append([]int{n % 10}, nums...)
		n = n / 10
	}
	idx := 0
	sort.Ints(digits)
	res := make([]int, len(nums))
	for idx < len(nums) {
		if findSame(nums[idx], digits) {
			res[idx] = nums[idx]
			idx++
		} else if findSmall(nums[idx], digits) {
			curNum := 0
			for i := 0; i < len(digits); i++ {
				if digits[i] < nums[idx] {
					curNum = max(curNum, digits[i])
				}
			}
			res[idx] = curNum
			idx++
			for idx < len(nums) {
				res[idx] = digits[len(digits)-1]
				idx++
			}
			break
		} else {
			// 回溯逻辑
			idx--
			for idx >= 0 && !findSmall(nums[idx], digits) {
				idx--
			}
			if idx < 0 {
				res[0] = 0
				for i := 1; i < len(res); i++ {
					res[i] = digits[len(digits)-1]
				}
			} else {
				curNum := 0
				for i := 0; i < len(digits); i++ {
					if digits[i] < nums[idx] {
						curNum = max(curNum, digits[i])
					}
				}
				res[idx] = curNum
				idx++
				for idx < len(nums) {
					res[idx] = digits[len(digits)-1]
					idx++
				}
			}
			break
		}
	}
	resNum := 0

	for i := 0; i < len(res); i++ {
		resNum = resNum*10 + res[i]
	}

	return resNum
}

package test

import (
	"testing"

	"go_coding_noalgorithm/src"
)

func TestBuildMaxLessThanN(t *testing.T) {
	tests := []struct {
		name   string
		n      int
		digits []int
		want   int
	}{
		{
			name:   "basic case",
			n:      345,
			digits: []int{1, 3, 5},
			want:   335,
		},
		{
			name:   "need backtrack in same length",
			n:      572,
			digits: []int{2, 5, 7},
			want:   557,
		},
		{
			name:   "fall back to shorter length",
			n:      1000,
			digits: []int{0, 1},
			want:   111,
		},
		{
			name:   "single digit only",
			n:      21,
			digits: []int{2},
			want:   2,
		},
		{
			name:   "zero is valid result",
			n:      10,
			digits: []int{0},
			want:   0,
		},
		{
			name:   "no solution",
			n:      1,
			digits: []int{9},
			want:   -1,
		},
		{
			name:   "n equals zero",
			n:      0,
			digits: []int{0, 1, 2},
			want:   -1,
		},
		{
			name:   "deduplicate digits",
			n:      332,
			digits: []int{3, 3, 1},
			want:   331,
		},
		{
			name:   "leading zero not allowed except zero itself",
			n:      100,
			digits: []int{0, 1},
			want:   11,
		},
		{
			name:   "empty digits",
			n:      999,
			digits: []int{},
			want:   -1,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			got := src.BuildMaxLessThanN(tt.n, tt.digits)
			if got != tt.want {
				t.Fatalf("n=%d digits=%v want=%d got=%d", tt.n, tt.digits, tt.want, got)
			}
		})
	}
}
